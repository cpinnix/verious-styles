import "./styles.css";

import { createBus } from "./utils/bus";
import { createRouterInteractor } from "./interactors/router";
import { createMainInteractor } from "./interactors/main";
import { createPresenterInteractor } from "./interactors/presenter";
import { createStateInteractor } from "./interactors/state";

const main = () => {
  const bus = createBus();

  const routerInteractor = createRouterInteractor();
  const mainInteractor = createMainInteractor();
  const stateInteractor = createStateInteractor();

  bus.register(routerInteractor);
  bus.register(mainInteractor);
  bus.register(stateInteractor);

  const presenter = createPresenterInteractor();

  bus.register(presenter);

  bus.dispatch({ name: "cmd.state.fetch" });
};

main();
